def Carbohidratos
        r = 0
        n = @alimentos.head
		i = @gramos.head


        while !n.nil?

			r += ((n.value.carbohidratos)*(i.value/100))
                n = n.next
			i = i.next
        end
        
		return (r*4).round(2)
end
