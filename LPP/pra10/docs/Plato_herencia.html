def new_GEI
	n=0
	r=0
	a = Array.new
	x = @gramos.head
	@alimentos.map{ |i|
		n = i.gei*(x.value/100)
		x = x.next
		a << 1.0 if n<800
		a << 2.0 if n>=800 && n<=1200
		a << 3.0 if n>1200

			 
	}
	r = self.Reduce(a)

	return (Float(r/@alimentos.Size)).round(1)
end
